open System

let alpha = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"

let convertText (text: string) : Double =
   text
   |> List.ofSeq
   |> List.rev
   |> List.mapi(
         fun i a -> 
            let power27 = Math.Pow(27., (float i))
            let alphaIndex = alpha.IndexOf(a) + 1 |> float
            let result = power27 * alphaIndex
            result)
   |> List.reduce(+)
