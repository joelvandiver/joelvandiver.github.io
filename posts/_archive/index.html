Printing Properties in FSI

Printing Properties of any given type to the FSI console for quick coding.

let printProperities<'T> () =
	typedefof<'T>
    	.GetProperties()
    	|> Seq.map(fun y -> y.Name)
    	|> Seq.iter(fun z -> printfn "%s" z)

type MyType =
  { PropA     : string
    PropB     : int
    PropC     : System.Guid  }

// FSI
> printProperities<MyType>();;
PropA
PropB
PropC
